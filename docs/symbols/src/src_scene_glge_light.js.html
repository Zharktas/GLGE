127 /**
128 * Gets the spot lights projection matrix
129 * @returns the lights spot projection matrix
130 * @private
131 */
132 GLGE.Light.prototype.getPMatrix=function(cvp,invlight,projectedDistance,distance){
133 	if(!this.spotPMatrix){
134 		var far;
135 		if(this.scene && this.scene.camera) far=this.scene.camera.far;
136 			else far=1000;
137 		if(this.type==GLGE.L_SPOT){
138 			this.spotPMatrix=GLGE.makePerspective(Math.acos(this.spotCosCutOff)/3.14159*360, 1.0, 0.1, far);
139 		}
140 	}
141 	 if(this.type==GLGE.L_DIR){
142 		this.spotPMatrix=GLGE.getDirLightProjection(cvp,invlight,projectedDistance,distance);
143 	}
144 	return this.spotPMatrix;
145 }

440 /**
441 * Sets up the WebGL needed to render the depth map for this light source. Only used for spot lights which produce shadows
442 * @private
443 */
444 GLGE.Light.prototype.createSpotBuffer=function(gl){
445     this.frameBuffer = gl.createFramebuffer();
446     this.renderBuffer = gl.createRenderbuffer();
447     this.texture = gl.createTexture();
448     gl.bindTexture(gl.TEXTURE_2D, this.texture);
449 
450     try {
451         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, this.bufferWidth, this.bufferHeight, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
452     } catch (e) {
453         GLGE.error("incompatible texture creation method");
454         var width=parseFloat(this.bufferWidth);
455         var height=parseFloat(this.bufferHeight);
456         var tex = new Uint8Array(width * height * 4);
457         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, width, height, 0, gl.RGBA, gl.UNSIGNED_BYTE, tex);
458     }
459     
460     gl.bindFramebuffer(gl.FRAMEBUFFER, this.frameBuffer);
461     gl.bindRenderbuffer(gl.RENDERBUFFER, this.renderBuffer);
462     gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, this.bufferWidth, this.bufferHeight);
463     
464     gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this.texture, 0);
465     gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this.renderBuffer);
466     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
467     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
468     gl.bindTexture(gl.TEXTURE_2D, null);
469 }

471 /**
472 * Sets up the buffers needed for the gaussian blured shadow buffer
473 * @private
474 */
475 GLGE.Light.prototype.createSoftBuffer=function(gl){
476     this.frameBufferSf = gl.createFramebuffer();
477     this.renderBufferSf = gl.createRenderbuffer();
478     this.textureSf = gl.createTexture();
479 480     gl.bindTexture(gl.TEXTURE_2D, this.textureSf);
481 
482     try {
483         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, this.bufferWidth, this.bufferHeight, 0, gl.RGBA, gl.UNSIGNED_BYTE, null);
484     } catch (e) {
485         GLGE.error("incompatible texture creation method");
486         var width=parseFloat(this.bufferWidth);
487         var height=parseFloat(this.bufferHeight);
488         var tex = new Uint8Array(width * height * 4);
489         gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, width, height, 0, gl.RGBA, gl.UNSIGNED_BYTE, tex);
490     }
491     
492     gl.bindFramebuffer(gl.FRAMEBUFFER, this.frameBufferSf);
493     gl.bindRenderbuffer(gl.RENDERBUFFER, this.renderBufferSf);
494     gl.renderbufferStorage(gl.RENDERBUFFER, gl.DEPTH_COMPONENT16, this.bufferWidth, this.bufferHeight);
495     
496     gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, this.texture, 0);
497     gl.framebufferRenderbuffer(gl.FRAMEBUFFER, gl.DEPTH_ATTACHMENT, gl.RENDERBUFFER, this.renderBufferSf);
498     gl.bindFramebuffer(gl.FRAMEBUFFER, null);
499     gl.bindRenderbuffer(gl.RENDERBUFFER, null);
500     gl.bindTexture(gl.TEXTURE_2D, null);
501     
502     //create the vertex positions
503 	if(!this.posBuffer) this.posBuffer = gl.createBuffer();
504 	gl.bindBuffer(gl.ARRAY_BUFFER, this.posBuffer);
505 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array([1,1,0,-1,1,0,-1,-1,0,1,-1,0]), gl.STATIC_DRAW);
506 	this.posBuffer.itemSize = 3;
507 	this.posBuffer.numItems = 4;
508 	//create the vertex uv coords
509 	if(!this.uvBuffer) this.uvBuffer = gl.createBuffer();
510 	gl.bindBuffer(gl.ARRAY_BUFFER, this.uvBuffer);
511 	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array([0,0,1,0,1,1,0,1]), gl.STATIC_DRAW);
512 	this.uvBuffer.itemSize = 2;
513 	this.uvBuffer.numItems = 4;
514 	//create the faces
515 	if(!this.GLfaces) this.GLfaces = gl.createBuffer();
516 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.GLfaces);
517 	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array([2,1,0,0,3,2]), gl.STATIC_DRAW);
518 	this.GLfaces.itemSize = 1;
519 	this.GLfaces.numItems = 6;
520 }

522 /**
523 * Sets up the programs require to do the soft shadows
524 * @private
525 */
526 GLGE.Light.prototype.createSoftPrograms=function(gl){
527 	if(this.GLShaderProgram) gl.deleteProgram(this.GLShaderProgram);
528 
529 	var vertexStr="";
530 	vertexStr+="attribute vec3 position;\n";
531 	vertexStr+="attribute vec2 uvcoord;\n";
532 	vertexStr+="varying vec2 texcoord;\n";
533 	vertexStr+="void main(void){\n";
534 	vertexStr+="texcoord=uvcoord;\n";    
535 	vertexStr+="gl_Position = vec4(position,1.0);\n";
536 	vertexStr+="}\n";
537 
538 	var fragStr="precision mediump float;\n";
539 	fragStr=fragStr+"uniform sampler2D TEXTURE;\n";
540 	fragStr=fragStr+"varying vec2 texcoord;\n";
541 	fragStr=fragStr+"float blurSize = 0.01;\n";
542 	fragStr=fragStr+"void main(void){\n";
543 	fragStr=fragStr+"vec4 color = vec4(0.0,0.0,0.0,0.0);";
544 	fragStr=fragStr+"float rnd = 1.0-rand(texCoord.xy)*4.0*blurSize;";
545 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x - 4.0*blurSize, texCoord.y)) * 0.05 * rnd;";
546 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x - 3.0*blurSize, texCoord.y)) * 0.09 * rnd;";
547 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x - 2.0*blurSize, texCoord.y)) * 0.12 * rnd;";
548 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x - blurSize, texCoord.y)) * 0.15 * rnd;";
549 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x, texCoord.y)) * 0.18 * rnd;";
550 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x + blurSize, texCoord.y)) * 0.15 * rnd;";
551 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x + 2.0*blurSize, texCoord.y)) * 0.12 * rnd;";
552 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x + 3.0*blurSize, texCoord.y)) * 0.09 * rnd;";
553 	fragStr=fragStr+"color += texture2D(TEXTURE, vec2(texCoord.x + 4.0*blurSize, texCoord.y)) * 0.05 * rnd;";
554 	fragStr=fragStr+"gl_FragColor = color;\n";
555 	fragStr=fragStr+"}\n";
556 
557 	this.GLFragmentShader=gl.createShader(gl.FRAGMENT_SHADER);
558 	this.GLVertexShader=gl.createShader(gl.VERTEX_SHADER);
559 
560 	gl.shaderSource(this.GLFragmentShader, fragStr);
561 	gl.compileShader(this.GLFragmentShader);
562 	if (!gl.getShaderParameter(this.GLFragmentShader, gl.COMPILE_STATUS)) {
563 	      GLGE.error(gl.getShaderInfoLog(this.GLFragmentShader));
564 	      return;
565 	}
566 
567 	gl.shaderSource(this.GLVertexShader, vertexStr);
568 	gl.compileShader(this.GLVertexShader);
569 	if (!gl.getShaderParameter(this.GLVertexShader, gl.COMPILE_STATUS)) {
570 		GLGE.error(gl.getShaderInfoLog(this.GLVertexShader));
571 		return;
572 	}
573 
574 	this.GLShaderProgram = gl.createProgram();
575 	gl.attachShader(this.GLShaderProgram, this.GLVertexShader);
576 	gl.attachShader(this.GLShaderProgram, this.GLFragmentShader);
577 	gl.linkProgram(this.GLShaderProgram);	
578 }

580 /**
581 * Renders the blured shadow
582 * @private
583 */
584 GLGE.Light.prototype.GLRenderSoft=function(gl){
585 	if(!this.gl){
586 		this.GLInit(gl);
587 	}	
588 	
589 	gl.bindFramebuffer(gl.FRAMEBUFFER, this.frameBufferSf);
590 
591 	if(gl.program!=this.GLShaderProgram){
592 		gl.useProgram(this.GLShaderProgram);
593 		gl.program=this.GLShaderProgram;
594 	}
595 
596 	var attribslot;
597 	for(var i=0; i<8; i++) gl.disableVertexAttribArray(i);
598 	attribslot=GLGE.getAttribLocation(gl,this.GLShaderProgram, "position");
599 
600 	gl.bindBuffer(gl.ARRAY_BUFFER, this.posBuffer);
601 	gl.enableVertexAttribArray(attribslot);
602 	gl.vertexAttribPointer(attribslot, this.posBuffer.itemSize, gl.FLOAT, false, 0, 0);
603 
604 	attribslot=GLGE.getAttribLocation(gl,this.GLShaderProgram, "uvcoord");
605 	gl.bindBuffer(gl.ARRAY_BUFFER, this.uvBuffer);
606 	gl.enableVertexAttribArray(attribslot);
607 	gl.vertexAttribPointer(attribslot, this.uvBuffer.itemSize, gl.FLOAT, false, 0, 0);
608 
609 	gl.activeTexture(gl["TEXTURE0"]);
610 	gl.bindTexture(gl.TEXTURE_2D, this.glTexture);
611 	GLGE.setUniform(gl,"1i",GLGE.getUniformLocation(gl,this.GLShaderProgram, "TEXTURE"),0);
612 
613 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.GLfaces);
614 	gl.drawElements(gl.TRIANGLES, this.GLfaces.numItems, gl.UNSIGNED_SHORT, 0);
615 	gl.bindFramebuffer(gl.FRAMEBUFFER, null);
616 }
