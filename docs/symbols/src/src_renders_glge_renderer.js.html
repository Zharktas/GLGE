 42 /**

 47 GLGE.Renderer=function(canvas,error,props){
 48 	this.viewport=[];
 49 	this.canvas=canvas;
 50 	if(!props) props={alpha:true,depth:true,stencil:true,antialias:true,premultipliedAlpha:true};
 51 	try {
 52 		this.gl = canvas.getContext("experimental-webgl",props);
 53 	} catch(e) {}
 54 	try {
 55 		if(!this.gl) this.gl = canvas.getContext("webgl",props);
 56 	} catch(e) {}
 57 	if(!this.gl) {
 58         console.log("GLGE err:", typeof(globalNoWebGLError)=="undefined")
 59 		if( (!error) && (typeof(globalNoWebGLError)=="undefined")){
 60 			var div=document.createElement("div");
 61 			div.setAttribute("style","position: absolute; top: 10px; left: 10px; font-family: sans-serif; font-size: 14px; padding: 10px;background-color: #fcffcb;color: #800; width: 200px; border:2px solid #f00");
 62 			div.innerHTML="WebGL compatible Browser Required(Firefox 4 or Chrome 9 and up) or you may need to update your graphics card driver.";
 63 			document.getElementsByTagName("body")[0].appendChild(div);
 64 			throw "cannot create webgl context";
 65 		}else{
 66 			error();
 67 			throw "cannot create webgl context";
 68 		}
 69 	}
 70 	//firefox is doing something here?
 71 	try{
 72 	this.gl.canvas=canvas;
 73 	}catch(e){};
 74 	//this.gl = WebGLDebugUtils.makeDebugContext(this.gl);
 75 	//this.gl.setTracing(true);
 76 
 77 	//chome compatibility
 78 	//TODO: Remove this when chome is right
 79 	if (!this.gl.getProgramParameter)
 80 	{
 81 		this.gl.getProgramParameter = this.gl.getProgrami
 82 	}
 83 	if (!this.gl.getShaderParameter)
 84 	{
 85 		this.gl.getShaderParameter = this.gl.getShaderi
 86 	}
 87 	// End of Chrome compatibility code
 88 	
 89 	this.gl.uniformMatrix4fvX=this.gl.uniformMatrix4fv
 90 	this.gl.uniformMatrix4fv=function(uniform,transpose,array){
 91 		if(!transpose){
 92 			this.uniformMatrix4fvX(uniform,false,array);
 93 		}else{
 94 			GLGE.mat4gl(GLGE.transposeMat4(array),array);
 95 			this.uniformMatrix4fvX(uniform,false,array);
 96 		}
 97 	}
 98 	var gl=this.gl;
 99 	
100 	/*this.gl.texImage2Dx=this.gl.texImage2D;
101 	this.gl.texImage2D=function(){
102 		if(arguments.length==9){
103 			gl.texImage2Dx(arguments[0], arguments[1], arguments[2],arguments[3],arguments[4],arguments[5],arguments[6],arguments[7],arguments[8]);
104 		}else{
105 			gl.texImage2Dx(arguments[0], arguments[1], arguments[5],false,false);
106 		}
107 	}*/
108 
109 	
110 	//set up defaults
111 	this.gl.clearDepth(1.0);
112 	this.gl.clearStencil(0);
113 	this.gl.enable(this.gl.DEPTH_TEST);
114     
115     
116 	this.gl.depthFunc(this.gl.LEQUAL);
117 	this.gl.blendFuncSeparate(this.gl.SRC_ALPHA,this.gl.ONE_MINUS_SRC_ALPHA,this.gl.ZERO,this.gl.ONE);	
118 };

212 /**
213 * Sets the clear type for rendering GLGE.C_ALL, GLGE.C_STENCIL, GLGE.C_DEPTH, GLGE.C_COLOR
214 * @param type how to clear the viewport for the next render
215 */
216 GLGE.Renderer.prototype.setClearType=function(type){
217 	this.clearType=type;
218 	return this;

220 /**
221 * Gets the clear type for rendering GLGE.C_ALL, GLGE.C_STENCIL, GLGE.C_DEPTH, GLGE.C_COLOR
222 * @returns how to clear the viewport for the next render
223 */
224 GLGE.Renderer.prototype.getClearType=function(){
225 	return this.clearType;
226 };
227 /**
228 * Clears the viewport
229 * @private
230 231 */
232 GLGE.Renderer.prototype.GLClear=function(){
233 	var gl=this.gl;
234 	var clearType=this.clearType;
235 	var clear=0;
236 	if((clearType & GLGE.C_COLOR) ==  GLGE.C_COLOR){
237 		clear=clear | gl.COLOR_BUFFER_BIT;
238 	}
239 	if((clearType & GLGE.C_DEPTH) == GLGE.C_DEPTH){
240 		clear=clear | gl.DEPTH_BUFFER_BIT;
241 	}
242 	if((clearType & GLGE.C_STENCIL) == GLGE.C_STENCIL){
243 		clear=clear | gl.STENCIL_BUFFER_BIT;
244 	}
245 	gl.clear(clear);
246 };

251 GLGE.Renderer.prototype.getScene=function(){
252 	return this.scene;
253 };
254 /**
255 * Sets the scene to render
256 * @param {GLGE.Scene} scene The scene to be rendered
257 */
258 GLGE.Renderer.prototype.setScene=function(scene){
259 	scene.renderer=this;
260 	this.scene=scene;
261 	if(!scene.gl) scene.GLInit(this.gl);
262 	//this.render();
263 	scene.camera.updateMatrix(); //reset camera matrix to force cache update
264 	return this;
265 };
266 /**
267 * Renders the current scene to the canvas
268 */
269 GLGE.Renderer.prototype.render=function(){
270 	if(this.cullFaces) this.gl.enable(this.gl.CULL_FACE);
271 	if (this.scene)
272 	this.scene.render(this.gl);
273 	//if this is the first ever pass then render twice to fill shadow buffers
274 	if(!this.rendered&&this.scene){
275 		this.scene.render(this.gl);
276 		this.rendered=true;
277 	}
278 };
